PHASE 1 — DB: Add only 3 new tables (do NOT touch any existing tables):

CREATE TABLE ai_conversation_context (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id UUID NOT NULL,
  tenant_id UUID NOT NULL,
  status TEXT DEFAULT 'active',  -- active | awaiting_price | awaiting_payment | confirmed
  context JSONB,  -- { destination, origin, departureDate, returnDate, passengers, flightClass }
  updated_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE ai_payments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID NOT NULL,
  conversation_id UUID,
  customer_phone TEXT,
  image_url TEXT,
  amount DECIMAL(10,2),
  currency TEXT DEFAULT 'SAR',
  vision_data JSONB,  -- { isReceipt, bankName, transactionId, date, confidence }
  status TEXT DEFAULT 'pending',  -- pending | approved | rejected
  reviewed_by TEXT,
  reviewed_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE ai_sla_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID NOT NULL,
  payment_id UUID,
  type TEXT,  -- customer_reassurance | admin_urgent
  sent_at TIMESTAMP DEFAULT NOW(),
  resolved BOOLEAN DEFAULT false
);
